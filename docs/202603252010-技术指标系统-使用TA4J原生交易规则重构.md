```java
/**
 * 创建MA策略
 */
private Strategy createMAStrategy(BarSeries series, int period) {
    ClosePriceIndicator closePrice = new ClosePriceIndicator(series);
    SMAIndicator sma = new SMAIndicator(closePrice, period);
    
    // 买入规则：价格上穿MA
    Rule entryRule = new CrossedUpIndicatorRule(closePrice, sma);
    
    // 卖出规则：价格下穿MA
    Rule exitRule = new CrossedDownIndicatorRule(closePrice, sma);
    
    return new BaseStrategy("MA" + period, entryRule, exitRule);
}

/**
 * 创建RSI策略
 */
private Strategy createRSIStrategy(BarSeries series, int period) {
    ClosePriceIndicator closePrice = new ClosePriceIndicator(series);
    RSIIndicator rsi = new RSIIndicator(closePrice, period);
    
    Num oversold = series.numFactory().numOf(30);
    Num overbought = series.numFactory().numOf(70);
    
    // 买入规则：RSI < 30（超卖）
    Rule entryRule = new UnderIndicatorRule(rsi, oversold);
    
    // 卖出规则：RSI > 70（超买）
    Rule exitRule = new OverIndicatorRule(rsi, overbought);
    
    return new BaseStrategy("RSI" + period, entryRule, exitRule);
}

/**
 * 创建MACD策略
 */
private Strategy createMACDStrategy(BarSeries series, int shortPeriod, int longPeriod, int signalPeriod) {
    ClosePriceIndicator closePrice = new ClosePriceIndicator(series);
    MACDIndicator macd = new MACDIndicator(closePrice, shortPeriod, longPeriod);
    EMAIndicator signal = new EMAIndicator(macd, signalPeriod);
    
    // 买入规则：MACD上穿信号线（金叉）
    Rule entryRule = new CrossedUpIndicatorRule(macd, signal);
    
    // 卖出规则：MACD下穿信号线（死叉）
    Rule exitRule = new CrossedDownIndicatorRule(macd, signal);
    
    return new BaseStrategy("MACD", entryRule, exitRule);
}
```

### 2. 执行策略获取信号

```java
/**
 * 使用TA4J策略生成交易信号
 */
private IndicatorResult generateSignalWithStrategy(
        BarSeries series,
        Strategy strategy,
        IndicatorConfig config,
        String exchange,
        Long timestamp) {
    
    int endIndex = series.getEndIndex();
    
    // 使用策略判断是否应该买入或卖出
    boolean shouldEnter = strategy.shouldEnter(endIndex);
    boolean shouldExit = strategy.shouldExit(endIndex);
    
    IndicatorResult result = new IndicatorResult();
    result.symbol = series.getName();
    result.exchange = exchange;
    result.indicatorType = config.indicatorType;
    result.indicatorName = config.indicatorName;
    result.timestamp = timestamp;
    result.createTime = new java.text.SimpleDateFormat("yyyy-MM-dd HH:mm:ss")
        .format(new Date(timestamp));
    result.extraValues = new HashMap<>();
    
    // 获取当前价格
    result.currentPrice = new BigDecimal(
        series.getBar(endIndex).getClosePrice().toString()
    );
    
    // 根据策略判断结果设置信号
    if (shouldEnter) {
        result.signalType = "BUY";
        result.signalReason = strategy.getName() + " entry signal";
        result.signalStrength = new BigDecimal("80");  // TA4J信号强度固定为80
    } else if (shouldExit) {
        result.signalType = "SELL";
        result.signalReason = strategy.getName() + " exit signal";
        result.signalStrength = new BigDecimal("80");
    } else {
        result.signalType = "HOLD";
        result.signalReason = "No signal";
        result.signalStrength = BigDecimal.ZERO;
    }
    
    return result;
}
```

### 3. 重构calculateIndicator方法

```java
private IndicatorResult calculateIndicator(
        BarSeries series,
        IndicatorConfig config,
        String exchange,
        Long timestamp) {
    
    if (series.getBarCount() < 2) {
        return null;
    }
    
    Strategy strategy = null;
    BigDecimal indicatorValue = null;
    Map<String, BigDecimal> extraValues = new HashMap<>();
    
    try {
        switch (config.indicatorType.toUpperCase()) {
            case "MA":
                int maPeriod = Integer.parseInt(config.params.getOrDefault("period", "20"));
                strategy = createMAStrategy(series, maPeriod);
                // 计算指标值
                ClosePriceIndicator closePrice = new ClosePriceIndicator(series);
                SMAIndicator sma = new SMAIndicator(closePrice, maPeriod);
                indicatorValue = new BigDecimal(sma.getValue(series.getEndIndex()).toString());
                break;
                
            case "EMA":
                int emaPeriod = Integer.parseInt(config.params.getOrDefault("period", "12"));
                strategy = createEMAStrategy(series, emaPeriod);
                // 计算指标值
                ClosePriceIndicator closePrice2 = new ClosePriceIndicator(series);
                EMAIndicator ema = new EMAIndicator(closePrice2, emaPeriod);
                indicatorValue = new BigDecimal(ema.getValue(series.getEndIndex()).toString());
                break;
                
            case "RSI":
                int rsiPeriod = Integer.parseInt(config.params.getOrDefault("period", "14"));
                strategy = createRSIStrategy(series, rsiPeriod);
                // 计算指标值
                ClosePriceIndicator closePrice3 = new ClosePriceIndicator(series);
                RSIIndicator rsi = new RSIIndicator(closePrice3, rsiPeriod);
                indicatorValue = new BigDecimal(rsi.getValue(series.getEndIndex()).toString());
                break;
                
            case "MACD":
                int shortPeriod = Integer.parseInt(config.params.getOrDefault("short_period", "12"));
                int longPeriod = Integer.parseInt(config.params.getOrDefault("long_period", "26"));
                int signalPeriod = Integer.parseInt(config.params.getOrDefault("signal_period", "9"));
                strategy = createMACDStrategy(series, shortPeriod, longPeriod, signalPeriod);
                // 计算指标值
                ClosePriceIndicator closePrice4 = new ClosePriceIndicator(series);
                MACDIndicator macd = new MACDIndicator(closePrice4, shortPeriod, longPeriod);
                EMAIndicator signal = new EMAIndicator(macd, signalPeriod);
                Num dif = macd.getValue(series.getEndIndex());
                Num dea = signal.getValue(series.getEndIndex());
                indicatorValue = new BigDecimal(dif.minus(dea).toString());
                extraValues.put("dif", new BigDecimal(dif.toString()));
                extraValues.put("dea", new BigDecimal(dea.toString()));
                break;
                
            // ... 其他指标类型
                
            default:
                LOG.warn("Unknown indicator type: {}", config.indicatorType);
                return null;
        }
        
        // 使用策略生成信号
        IndicatorResult result = generateSignalWithStrategy(
            series, strategy, config, exchange, timestamp
        );
        result.indicatorValue = indicatorValue;
        result.extraValues = extraValues;
        
        return result;
        
    } catch (Exception e) {
        LOG.error("Error calculating indicator: {}", config.indicatorName, e);
        return null;
    }
}
```

```java
private Strategy createMAWithRSIStrategy(BarSeries series) {
    ClosePriceIndicator closePrice = new ClosePriceIndicator(series);
    SMAIndicator sma = new SMAIndicator(closePrice, 20);
    RSIIndicator rsi = new RSIIndicator(closePrice, 14);
    
    // 买入规则：价格上穿MA 且 RSI < 30

    return new BaseStrategy("MA20+RSI14", entryRule, exitRule);
}
```

## 技术要点

### 1. TA4J规则系统

**规则接口**：
```java
public interface Rule {
    boolean isSatisfied(int index);
    boolean isSatisfied(int index, TradingRecord tradingRecord);
}
```

**常用规则**：
- `CrossedUpIndicatorRule`: 指标上穿
- `CrossedDownIndicatorRule`: 指标下穿
- `OverIndicatorRule`: 指标高于阈值
- `UnderIndicatorRule`: 指标低于阈值
- `StopLossRule`: 止损规则
- `StopGainRule`: 止盈规则

### 2. 策略组合

**AND组合**：
```java
Rule entryRule = rule1.and(rule2).and(rule3);
// 所有规则都满足才触发
```

**OR组合**：
```java
Rule exitRule = rule1.or(rule2);
// 任一规则满足就触发
```

**投票规则**：
```java
Rule voteRule = new VoteRule(2, rule1, rule2, rule3);
// 至少2个规则满足才触发
```

### 3. 信号强度

TA4J的规则返回boolean值（满足/不满足），没有信号强度概念。我们可以：

**方案1：固定强度**
- BUY信号：80
- SELL信号：80
- HOLD：0

**方案2：根据指标值计算强度**
```java
// RSI信号强度
if (rsi < 30) {
    strength = (30 - rsi) * 3;  // 0-90
} else if (rsi > 70) {
    strength = (rsi - 70) * 3;  // 0-90
}
```
